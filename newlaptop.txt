function Post-Json($url, $obj) {
  $json  = ($obj | ConvertTo-Json -Depth 50 -Compress)
  $bytes = [System.Text.Encoding]::UTF8.GetBytes($json)
  return Invoke-RestMethod -Method Post -Uri $url -Headers $H -Body $bytes
}

# 2) Try every package for Main:Device and Main:BrokerContract
$hits = @()

foreach ($p in $pkgs) {
  $tDev = "$p`:Main:Device"
  $tBC  = "$p`:Main:BrokerContract"

  try {
    $rDev = Post-Json "$JSON_API/v1/query" @{ templateIds = @($tDev) }
    $cDev = @($rDev.result).Count
  } catch { $cDev = -1 }

  try {
    $rBC = Post-Json "$JSON_API/v1/query" @{ templateIds = @($tBC) }
    $cBC = @($rBC.result).Count
  } catch { $cBC = -1 }

  if ($cDev -gt 0 -or $cBC -gt 0) {
    $hits += [pscustomobject]@{ packageId=$p; Devices=$cDev; BrokerContracts=$cBC }
  }
}
